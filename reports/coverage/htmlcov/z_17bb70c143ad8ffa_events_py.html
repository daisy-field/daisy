21class Event: 

22 """Specific event, with a start and an end timestamp, its label, and conditions 

23 whether a data point should be labeled as such (function evaluating to true if 

24 that is the case, false otherwise). 

25 """ 

26 

27 start_time: datetime 

28 end_time: datetime 

29 label: str 

30 _condition_fn: Callable[[list[dict]], bool] 

31 

32 def __init__( 

33 self, 

34 start_time: datetime, 

35 end_time: datetime, 

36 label: str, 

37 condition_fn: Callable[[list[dict]], bool], 

38 ): 

39 """Creates an event, which takes place between start_time and end_time. Data 

40 points evaluated true by the condition function should be labeled with the 

41 provided label. 

42 

43 :param start_time: Start time of event. 

44 :param end_time: End time of event. 

45 :param label: Label by which data points falling in the event should be labeled. 

46 :param condition_fn: Condition function used to evaluate if a given data point 

47 falls within the event. 

48 """ 

49 self.start_time = start_time 

50 self.end_time = end_time 

51 self.label = label 

52 self._condition_fn = condition_fn 

53 

54 def evaluate(self, timestamp: datetime, data: list[dict]) -> bool: 

55 """Evaluates a single data point using the condition function provided in the 

56 constructor. The timestamp is used to determine whether the data point falls 

57 within the events time frame. Additional meta information can be provided by 

58 passing multiple dictionaries in the data parameter. The dictionaries will be 

59 searched in the provided order. The first value found will be used for 

60 comparisons in the condition function. 

61 

62 :param timestamp: The timestamp of the data point. 

63 :param data: A single data point and additional meta information. The data 

64 is searched in the provided order. E.g. if two dictionaries contain the key 

65 timestamp and the condition function uses this key, then the value of the first 

66 dictionary will be used. 

67 """ 

68 if self.start_time <= timestamp <= self.end_time: 

69 return self._condition_fn(data) 

70 else: 

71 return False

74class EventParser: 

75 """Parser for conditions of events. It takes an expression, parses it, and returns 

76 a function, which evaluates if a given data point fulfils the condition. 

77 

78 The condition has to follow the following grammar: 

79 exp := pars + (binary_op + pars)? | 

80 unary_op + pars 

81 pars := operand | '(' + exp + ')' 

82 operand := word + comparator + word 

83 word := [any character except [] !"'<=>\\()] | 

84 '[' + [any character except []!"'<=>\\()] + ']' Note that whitespaces are 

85 allowed with brackets 

86 comparator := '=' | 'in' 

87 binary_op := 'and' | 'or' 

88 unary_op := 'not' 

89 

90 For comparators, the feature in the dictionary is always expected on the left side 

91 of the comparator, except with the 'in' operator, where it is expected on the right. 

92 

93 Some example expressions are: 

94 ip.addr = 10.1.1.1 When the function is called with a dictionary, it will 

95 be searched for the key ip.addr. Its value will be 

96 compared to 10.1.1.1 

97 tcp in protocols The dictionary will be searched for the key protocols. 

98 The function 'tcp in <value of protocols>' will be 

99 evaluated. 

100 

101 ip.addr = 10.1.1.1 and tcp in protocols 

102 (ip.addr = 10.1.1.1 or ip.addr = 192.168.1.1) and tcp in protocols 

103 not (ip.addr = 10.1.1.1 or ip.addr = 192.168.1.1) and tcp in protocols 

104 

105 The returned function can be called using a list of dictionaries. The dictionaries 

106 will be searched in the provided order and the first occurrence of a feature in one 

107 of the dictionaries will be used. This can be used to provide meta information about 

108 a data point additionally to the data point itself. 

109 """ 

110 

111 # The comparators and operators used by the parser 

112 _comparators: list[str] = ["=", "in"] 

113 _binary_operators: list[str] = ["and", "or"] 

114 _unary_operators: list[str] = ["not"] 

115 

116 # Identifiers used inside the parser to mark specific tokens 

117 _var1: str = "var1" 

118 _var2: str = "var2" 

119 _op: str = "op" 

120 

121 _parser: pp.ParserElement 

122 

123 def __init__(self): 

124 """Creates the grammar for the conditions. The specific grammar can be seen 

125 in the description of this class. 

126 """ 

127 pp.ParserElement.enablePackrat() 

128 sys.setrecursionlimit(3000) 

129 

130 base_word = pp.Word(pp.printables, exclude_chars="[] !\"'<=>\\()") 

131 bracket_word = pp.Word(pp.printables + " ", exclude_chars="[]!\"'<=>\\()") 

132 

133 comparator = pp.oneOf(self._comparators) 

134 boperator = pp.oneOf(self._binary_operators) 

135 uoperator = pp.oneOf(self._unary_operators) 

136 lpar = pp.Suppress("(") 

137 rpar = pp.Suppress(")") 

138 lbr = pp.Suppress("[") 

139 rbr = pp.Suppress("]") 

140 

141 self._parser = pp.Forward() 

142 word = base_word | lbr + bracket_word + rbr 

143 operand = pp.Group(word(self._var1) + comparator(self._op) + word(self._var2)) 

144 pars = operand | lpar + self._parser + rpar 

145 

146 self._parser <<= pp.Group( 

147 pars(self._var1) 

148 + pp.Optional(boperator(self._op) + self._parser(self._var2)) 

149 ) | pp.Group(uoperator(self._op) + pars(self._var1)) 

150 

151 def parse(self, expression: str) -> Callable[[list[dict]], bool]: 

152 """Parses the given expression and returns a function that evaluates data points 

153 passed to it. This method can raise a parse error if the expression is invalid. 

154 

155 :param expression: Expression (condition) to parse. 

156 :raises ParseError: Any condition/expression does not follow parser's grammar. 

157 """ 

158 tree = self._parser.parseString(expression, parseAll=True) 

159 return self._process_child(tree) 

160 

161 def _process_child( 

162 self, tree: pp.ParseResults 

163 ) -> Optional[Callable[[list[dict]], bool]]: 

164 """Recursively processes the provided parse tree and returns a function that 

165 evaluates the data points passed to it. 

166 

167 :param tree: The parse tree to process 

168 """ 

169 result = {} 

170 if isinstance(tree, pp.ParseResults): 

171 # Evaluate all children first 

172 if tree.haskeys(): 

173 for key, _ in reversed(tree.asDict().items()): 

174 result[key] = self._process_child(tree.pop()) 

175 else: 

176 return self._process_child(tree.pop()) 

177 else: 

178 # On leaf node do nothing 

179 return None 

180 

181 return self._create_fn(tree.asDict(), result)

183 def _create_fn( 

184 self, dictionary: dict, result: dict 

185 ) -> Callable[[list[dict]], bool]: 

186 """Creates a function based on the provided input. The provided dictionary 

187 should contain any of the keys _op, _var1, and _var2 from this class. 

188 When the key _op is any of the binary operators, _var1 and _var2 have to be 

189 present in the result dictionary and contain a function. If the operator is 

190 any unary operator, only _var1 needs to be present in the result dictionary. 

191 If the operator is any comparator, the result dictionary is ignored and the 

192 dictionary needs to contain the _var1 and _var2 keys. 

193 The returned function can raise key errors if the dictionaries passed to it 

194 do not contain the keys required by the generated function. 

195 

196 :param dictionary: Dictionary used to evaluate the function to generate. 

197 :param result: Dictionary containing results of previous runs of this function. 

198 :raises NotImplementedError: Operators used are unsupported by this class. 

199 Only happens when class has been modified/overridden. 

200 """ 

201 if self._op not in dictionary: 

202 return list(result.values())[0] 

203 

204 operation = dictionary[self._op] 

205 

206 if operation in self._binary_operators: 

207 return self._create_binary_fn(result, operation) 

208 

209 if operation in self._unary_operators: 

210 return self._create_unary_fn(result, operation) 

211 

212 if operation in self._comparators: 

213 return self._create_comparator_fn(dictionary, operation) 

214 

215 raise NotImplementedError( 

216 f"Operation {operation} not supported in EventParser." 

217 ) 

218 

219 def _create_binary_fn( 

220 self, result: dict, operation: str 

221 ) -> Callable[[list[dict]], bool]: 

222 """Creates the binary function part. The results of previous runs of the 

223 create_fn method are used in this function to combine them using binary 

224 functions (and, or operators). 

225 

226 :param result: Dictionary containing the results of previous runs. 

227 :param operation: Operation to perform. 

228 :raises NotImplementedError: Operators used are unsupported by this class. 

229 Only happens when class has been modified/overridden. 

230 """ 

231 match operation: 

232 case "and": 

233 return lambda data: result[self._var1](data) and result[self._var2]( 

234 data 

235 ) 

236 case "or": 

237 return lambda data: result[self._var1](data) or result[self._var2](data) 

238 case _: 

239 raise NotImplementedError( 

240 f"Operation {operation} not supported in EventParser." 

241 ) 

242 

243 def _create_unary_fn( 

244 self, result: dict, operation: str 

245 ) -> Callable[[list[dict]], bool]: 

246 """Creates the unary function part. The results of previous runs of the 

247 create_fn method are used in this function to combine them using unary 

248 functions (not operator). 

249 

250 :param result: Dictionary containing the results of previous runs 

251 :param operation: Operation to perform. 

252 :raises NotImplementedError: Operators used are unsupported by this class. 

253 Only happens when class has been modified/overridden. 

254 """ 

255 match operation: 

256 case "not": 

257 return lambda data: not result[self._var1](data) 

258 case _: 

259 raise NotImplementedError( 

260 f"Operation {operation} not supported in EventParser." 

261 ) 

262 

263 def _create_comparator_fn( 

264 self, dictionary: dict, operation: str 

265 ) -> Callable[[list[dict]], bool]: 

266 """Creates the comparisons of the function. The dictionary should contain 

267 a feature, operation, and value. Based on the comparator, a function is 

268 returned, which performs the desired comparison. 

269 

270 :param dictionary: Dictionary containing feature and value to use for the 

271 generated function. 

272 :param operation: Operation to perform. 

273 :raises NotImplementedError: Operators used are unsupported by this class. 

274 Only happens when class has been modified/overridden. 

275 """ 

276 match operation: 

277 case "=": 

278 return ( 

279 lambda data: _get_value(dictionary[self._var1][0], data) 

280 == dictionary[self._var2][0] 

281 ) 

282 case "in": 

283 return ( 

284 lambda data: dictionary[self._var1][0] 

285 in _get_value(dictionary[self._var2][0], data) 

286 if _get_value(dictionary[self._var2][0], data) 

287 else False 

288 ) 

289 case _: 

290 raise NotImplementedError( 

291 f"Operation {operation} not supported in EventParser." 

292 )

295class EventHandler: 

296 """Event handler used to create events and automatically label data points. Events 

297 can be added to the class using the add_event() method. The events are matched to 

298 data points in added order, i.e. first matching event will be used to label 

299 a given data point. 

300 """ 

301 

302 _parser: EventParser 

303 _events: list[Event] 

304 _default_label: str 

305 _label_feature: str 

306 _error_label: str 

307 _hide_errors: bool 

308 

309 def __init__( 

310 self, 

311 default_label: str = "benign", 

312 label_feature: str = "label", 

313 error_label: str = "error", 

314 hide_errors: bool = False, 

315 name: str = "", 

316 ): 

317 """Creates an event handler used to label data points. 

318 

319 :param default_label: Label used when no event matches data point. 

320 :param label_feature: Feature in data point for which label will be set. 

321 :param error_label: Error label to use if an error is encountered during 

322 processing. 

323 :param hide_errors: Catches any key errors occurring in process() method when 

324 encountering data points not containing features used by the conditions of 

325 an event, only printing them out in the logs instead of exciting, labeling 

326 the data point as erroneous. 

327 :param name: Name of event handler for logging purposes. 

328 """ 

329 self._logger = logging.getLogger(name) 

330 

331 self._parser = EventParser() 

332 self._events = [] 

333 self._default_label = default_label 

334 self._label_feature = label_feature 

335 self._error_label = error_label 

336 self._hide_errors = hide_errors 

337 

338 def add_event( 

339 self, start_time: datetime, end_time: datetime, label: str, condition: str = "" 

340 ) -> Self: 

341 """Adds an event to the event handler. The events will be evaluated in the 

342 order they are provided. Each event has a start and end time, a label that will 

343 be used to label data points that fall under that event, and an optional 

344 condition. The condition is a string and has to follow a certain grammar: 

345 

346 exp := pars + (binary_op + pars)? | 

347 unary_op + pars 

348 pars := operand | '(' + exp + ')' 

349 operand := word + comparator + word 

350 word := [any character except [] !"'<=>\\()] | 

351 '[' + [any character except []!"'<=>\\()] + ']' Note that whitespaces 

352 are allowed with 

353 brackets 

354 comparator := '=' | 'in' 

355 binary_op := 'and' | 'or' 

356 unary_op := 'not' 

357 

358 For comparators, the feature in the dictionary is always expected on the left 

359 side of the comparator, except with the 'in' operator, where it is expected 

360 on the right. 

361 

362 Some example expressions are: 

363 ip.addr = 10.1.1.1 When the function is called with a dictionary, it will 

364 be searched for the key ip.addr. Its value will be 

365 compared to 10.1.1.1 

366 tcp in protocols The dictionary will be searched for the key protocols. 

367 The function 'tcp in <value of protocols>' will be 

368 evaluated. 

369 

370 Concatenation examples are: 

371 ip.addr = 10.1.1.1 and tcp in protocols 

372 (ip.addr = 10.1.1.1 or ip.addr = 192.168.1.1) and tcp in protocols 

373 not (ip.addr = 10.1.1.1 or ip.addr = 192.168.1.1) and tcp in protocols 

374 

375 The returned function can be called using a list of dictionaries. The 

376 dictionaries will be searched in the provided order and the first occurrence 

377 of a feature in one of the dictionaries will be used. This can be used to 

378 provide meta information about a data point additionally to the data point 

379 itself. 

380 

381 :param start_time: Start time of event. 

382 :param end_time: End time of event. 

383 :param label: Label of event. 

384 :param condition: Condition(s) data points have to fulfill for this event. 

385 """ 

386 self._logger.debug(f"Adding new event with condition {condition}") 

387 if condition: 

388 self._events.append( 

389 Event(start_time, end_time, label, self._parser.parse(condition)) 

390 ) 

391 else: 

392 self._events.append(Event(start_time, end_time, label, lambda _: True)) 

393 return self

395 def process( 

396 self, 

397 timestamp: datetime, 

398 data_point: dict, 

399 meta_data: list[dict] = None, 

400 ) -> dict: 

401 """Iterates through all events and checks for each event if it applies to 

402 the provided data point. If it does, the data point will be labeled with the 

403 label provided by the event. If no event matches the data point, it will be 

404 labeled with the default label. 

405 

406 :param timestamp: Timestamp of data point. 

407 :param data_point: Data point to label. 

408 :param meta_data: Additional meta information to label data point. Has 

409 preference over data point when checking conditions. 

410 processing and errors are suppressed. 

411 :return: Labelled data point. 

412 :raises KeyError: Data points does not contain feature used by a conditions and 

413 errors are not suppressed, i.e. redirected to log + data point is assigned 

414 the error label. 

415 """ 

416 if meta_data is None: 

417 meta_data = [] 

418 data = meta_data + [data_point] 

419 

420 for event in self._events: 

421 try: 

422 if event.evaluate(timestamp, data): 

423 data_point[self._label_feature] = event.label 

424 return data_point 

425 except KeyError as e: 

426 if not self._hide_errors: 

427 raise e 

428 self._logger.error(f"Error while evaluating event: {e}") 

429 data_point[self._label_feature] = self._error_label 

430 return data_point 

431 data_point[self._label_feature] = self._default_label 

432 return data_point

435def _get_value(feature: str, data: list[dict]) -> object: 

436 """Iterates through the provided dictionaries and returns the value of the first 

437 occurrence of the provided feature. 

438 

439 :param feature: The feature to find. 

440 :param data: List of dictionaries to search for the feature. 

441 """ 

442 for dictionary in data: 

443 if feature in dictionary: 

444 return dictionary[feature] 

445 raise KeyError(f"Could not find {feature} in {data}")
